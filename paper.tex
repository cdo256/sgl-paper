\usetikzlibrary{arrows.meta}

% Define a custom arrow tip
\makeatletter
\pgfarrowsdeclare{orthogonal}{orthogonal}
{
  \pgfutil@tempdima=0.5pt%
  \advance\pgfutil@tempdima by0.5\pgflinewidth%
  \pgfarrowsleftextend{-\pgfutil@tempdima}
  \pgfarrowsrightextend{\pgfutil@tempdima}
}
{
  \pgfpathmoveto{\pgfqpoint{-3pt}{-3pt}}
  \pgfpathlineto{\pgfqpoint{3pt}{3pt}}
  \pgfusepathqstroke
}
\makeatother

\begin{document}
\begin{tikzpicture}

% Use the custom arrow tip
\draw[->, orthogonal] (0,0) -- (2,0);

% Example graph with custom arrows
\node (a) at (0,0) {a};
\node (b) at (2,1) {b};
\node (c) at (2,-1) {c};
\node (d) at (4,0) {d};

\draw[orthogonal-] (a) -- (b);
\draw[orthogonal-] (a) -- (c);
\draw[->] (b) -- (d);
\draw[->] (c) -- (d);

\end{tikzpicture}
\end{document}
